\chapter*{Lista symboli}

\begin{itemize}[noitemsep,topsep=0pt,parsep=0pt,partopsep=0pt,labelwidth=1cm,align=left,itemindent=0pt]
\item[\(v\)] - całkowita wartość pobudzenia neuronu
\item[\(\mathbf{W}\)] - wektor wag sieci neuronowej
\item[\(\mathbf{X}\)] - wektor wejściowy
\item[\(b\)] - bias (składowa stała)
\item[\(x_i\)] - \textit{i}-te wejście
\item[\(w_i\)] - waga przypisana \textit{i}-temu wejściu
\item[\(y\)] - wartość wyjściowa neuronu
\item[\(f(\cdot)\)] - funkcja aktywacji
\item[\(\beta\)] - parametr odpowiadający za kształt charakterystyki sigmoidalnej funkcji aktywacji
\item[\(\Delta\)] - opóźnienie wyjścia podane na wejście w następnym kroku sieci rekurencyjnej.
\item[\(n\)] - liczba wymiarów danych
\item[\(X\)] - wektor wejściowy
\item[\(D(X, W_j)\)] - odległość między wektorem wejściowym a wektorem wag \(j\)-tego neuronu w sieci SOM
\item[\(W_j\)] - \textit{n}-wymiarowy wektor wag \(j\)-tego neuronu
\item[\(W_j(l)\)] - wektor wag neuronu w iteracji \(l\)
\item[\(\eta(k)\)] - współczynnik uczenia w epoce \(k\)
\item[\(R\)] - promień sąsiedztwa
\item[\(d(i, j)\)] - odległość topologiczna między neuronami
\item[\(G(R, d(i, j))\)] - wartość funkcji sąsiedztwa
\item[\(p_i(l)\)] - potencjał neuronu stosowany w algorytmie WTA
\item[\(p_{\text{min}}\)] - minimalny potencjał uprawniający do wzięcia udziału w rywalizacji podczas algorytmu WTA
\item[\(h\)] - indeks przypisywany podczas algorytmu gazu neuronowego
\item[\(G(f)\)] - funkcja sąsiedztwa w algorytmie gazu neuronowego
\item[\(\lambda\)] - parametr z algorytmu gazu neuronowego odpowiadający promieniowi sąsiedztwa z WTM
\item[\(X_i\)] - pojedynczy wzorzec wejściowy
\item[\(Q(X_i)\)] - błąd kwantyzacji dla pojedynczego neuronu
\item[\(Q_{\text{error}}\)] - średni błąd kwantyzacji, nazywany w pracy błędem kwantyzacji
\item[\(N\)] - liczba wektorów wejściowych w zbiorze danych
\item[\(Q_{\text{max}}\)] - maksymalny błąd kwantyzacji dla danego zbioru danych
\item[\(T_{\text{error}}\)] - błąd topograficzny
\item[\(h_k\)] - funkcje zmiany parametrów sieci SOM
\item[\(A\)] - początkowa wartość promienia sąsiedztwa/współczynnika uczenia, w funkcji \(h_k\)
\item[\(T\)] - całkowita liczba epok treningowych
\item[\(C\)] - stała czasowa w funkcji zmiany parametrów
\end{itemize}
